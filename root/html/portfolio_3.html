def isPrime(n):
prime = [True for i in range(n+1)]
p = 2
while p*p<=n:
    if prime[p]==True:
        for i in range(p*p,n+1,p):
            prime[i]=False
    p+=1

return prime[n]


def gcd(a,b):
while b!=0:
    r = a%b
    a=b
    b=r
return a

def Multiplicative_inverse(a,b):
s1 = 1
s2 = 0
m = b
while b!=0:
    q=a//b
    r=a%b
    a=b
    b=r
    s=s1-q*s2
    s1=s2
    s2=s

if s1<0:
    s1+=m

return s1

def powermod(x,y,p):
res = 1

x = x%p
while (y>0):

    if (y%2) == 1:
        res = (res*x)%p

    y = y//2
    x = (x*x)%p

return res

if __name__ == '__main__':
while (True):

            p = 13
            p = int(input('Enter a prime number: '))
            if isPrime(p):
                break

                break

                print('Both the prime numbers are same!! or product of both the prime numbers is less than 26!!')

        n = p*q
        phi_n = (p-1)*(q-1)
        a = 19

            a = int(input('Enter a number such that Greatest Common Divisor of that number with '+ str(phi_n) + ' is 1: '))

        b = Multiplicative_inverse(a,phi_n)
